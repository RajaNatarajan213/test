endpoint: customapp-nonprod.cqtkutzfwkzj.ap-southeast-1.rds.amazonaws.com
port : 4406

Connect to the database
-------------------------------------------------------------------------------------
mysql -h <endpoint> -P 4406 -u username -p 

-------------------------------------------------------------------------------------

Create user
-------------------------------------------------------------------------------------
CREATE USER 'username'@'%' IDENTIFIED BY 'password'

-------------------------------------------------------------------------------------

Grant privileges to a user for a database/schema_name

GRANT ALL PRIVILEGES ON schema_name.* TO 'username'@'%'

GRANT CREATE, ALTER, DROP, INSERT, UPDATE, DELETE, SELECT, REFERENCES, RELOAD on schema_name TO 'username'@'%' WITH GRANT OPTION;
-------------------------------------------------------------------------------------

Revoke Privileges for a user
-------------------------------------------------------------------------------------
REVOKE ALL ON schema_name FROM 'username'@'%';

REVOKE ALL PRIVILEGES ON *schema_name* FROM 'username'@'%';
-------------------------------------------------------------------------------------

Export database
-------------------------------------------------------------------------------------
Export single database
mysqldump -h <endpoint> -P 4406 -u username -p <schema_name> > <filename>.sql

-------------------------------------------------------------------------------------
Export all databases
mysqldump -h <endpoint> -P 4406 -u username -p --all-database > <filename>.sql

-------------------------------------------------------------------------------------

Restore from Database
-------------------------------------------------------------------------------------
Restore single database/schema

mysql -h <endpoint> -P 4406 -u username -p <schema_name> < <filename>.sql

-------------------------------------------------------------------------------------
Restore all database/schema

mysql -h <endpoint> -P 4406 -u username -p < <filename>.sql
